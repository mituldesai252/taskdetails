pipeline {
    agent any
    stages {
        stage('Fetch All Branches') {
            steps {
                sh '''
                cd /var/www/lms-projects/lmsdev
                git fetch --all --prune
                '''
            }
        }
        stage('Pull All Branches') {
            steps {
                sh '''
                cd /var/www/lms-projects/lmsdev
                for branch in $(git branch -r | grep -v "\\->" | sed "s/origin\///"); do
                    echo "Pulling branch: $branch"
                    git checkout $branch
                    git pull origin $branch
                done
                '''
            }
        }
    }
}
